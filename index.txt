  async function startCapture() {
    if (!navigator.mediaDevices) {
      showToast('ไม่รองรับ', 'กรุณาใช้ Chrome หรือ Edge', 'error');
      return;
    }

    try {
      const startRes = await apiPost({
        action: 'startSession',
        mode: modeSelect.value,
        sourceLang: sourceLang.value,
        targetLang: targetLang.value,
        saveMode: saveMode.value
      });

      if (startRes.error) throw new Error(startRes.error);
      session = startRes;

      if (session.folderUrl) {
        openFolderLink.href = session.folderUrl;
        openFolderLink.classList.remove('hidden');
      } else {
        openFolderLink.href = '#';
        openFolderLink.classList.add('hidden');
      }

      if (modeSelect.value === 'screen') {
        mediaStream = await navigator.mediaDevices.getDisplayMedia({
          video: true,
          audio: {
            echoCancellation: false,
            noiseSuppression: false,
            autoGainControl: false
          }
        });
      } else {
        mediaStream = await navigator.mediaDevices.getUserMedia({
          audio: {
            echoCancellation: true,
            noiseSuppression: true,
            autoGainControl: true
          }
        });
      }

      const audioTracks = mediaStream.getAudioTracks();
      if (!audioTracks.length) {
        throw new Error('ไม่พบเสียง ถ้าเลือกเสียงในเครื่องต้องติ๊ก Share audio / System audio');
      }

      mediaStream.getTracks().forEach(t => {
        t.onended = () => {
          if (isRecording) stopCapture();
        };
      });

      isRecording = true;
      chunkIndex = 0;
      fullAudioChunks = [];

      if (shouldSaveMedia()) startFullRecording();
      else log('โหมดนี้ไม่บันทึกไฟล์เสียงเต็มลง Drive');
      startChunkLoop();

      updateUIState();
      showToast('เริ่มแล้ว', 'กำลังถอดตามโหมด: ' + getSaveModeLabel(), 'success');
      log('สร้าง session: ' + session.sessionId);
      log('โหมดบันทึก: ' + getSaveModeLabel());
      if (session.folderUrl) log('Drive: ' + session.folderUrl);

    } catch (err) {
      showToast('เริ่มไม่สำเร็จ', err.message || String(err), 'error');
      stopCapture();
    }
  }

  function startFullRecording() {
    if (!mediaStream) return;

    const audioTracks = mediaStream.getAudioTracks();
    if (!audioTracks.length) return;

    const audioOnlyStream = new MediaStream(audioTracks);

    try {
      fullRecorder = new MediaRecorder(audioOnlyStream);
      fullAudioMimeType = fullRecorder.mimeType || 'audio/webm';
    } catch (err) {
      log('Full Recorder ERROR: ' + (err.message || String(err)));
      return;
    }

    fullRecorder.ondataavailable = e => {
      if (e.data && e.data.size > 0) fullAudioChunks.push(e.data);
    };

    fullRecorder.start(1000);
    log('เริ่มบันทึกเสียงเต็มในหน่วยความจำชั่วคราว');
  }

  function startChunkLoop() {
    const ms = Number(chunkSeconds.value || 15) * 1000;
    recordOneChunk(ms);
    chunkTimer = setInterval(() => {
      if (isRecording) recordOneChunk(ms);
    }, ms + 700);
  }

  function recordOneChunk(ms) {
    if (!mediaStream || !isRecording || !session) return;
    if (chunkRecorder && chunkRecorder.state === 'recording') return;

    const audioTracks = mediaStream.getAudioTracks();
    if (!audioTracks.length) return;

    const audioOnlyStream = new MediaStream(audioTracks);
    const chunks = [];

    try {
      chunkRecorder = new MediaRecorder(audioOnlyStream);
    } catch (err) {
      log('Recorder ERROR: ' + (err.message || String(err)));
      return;
    }

    chunkRecorder.ondataavailable = e => {
      if (e.data && e.data.size > 0) chunks.push(e.data);
    };

    chunkRecorder.onstop = async () => {
      try {
        const blob = new Blob(chunks, { type: chunkRecorder.mimeType || 'audio/webm' });

        if (blob.size < 1200) {
          log('ข้าม chunk เพราะเสียงสั้น/ไม่มีข้อมูล');
          return;
        }

        chunkIndex++;
        log('ส่ง chunk #' + chunkIndex + ' (' + Math.round(blob.size / 1024) + ' KB)');
        const audioBase64 = await blobToBase64(blob);

        const json = await apiPost({
          action: 'processChunk',
          sessionId: session.sessionId,
          folderId: session.folderId,
          transcriptFileId: session.transcriptFileId,
          translationFileId: session.translationFileId,
          audioBase64,
          mimeType: blob.type || 'audio/webm',
          sourceLang: sourceLang.value,
          targetLang: targetLang.value,
          chunkIndex,
          saveMode: saveMode.value
        });

        if (json.error) {
          log('API ERROR #' + chunkIndex + ': ' + json.error);
          showToast('API Error', json.error, 'error');
          return;
        }

        appendTranscript(json.transcript || '');
        appendTranslation(json.translation || '');
        log('chunk #' + chunkIndex + ' บันทึกแล้ว');

      } catch (err) {
        log('Chunk ERROR: ' + (err.message || String(err)));
      }
    };

    try {
      chunkRecorder.start();
    } catch (err) {
      log('Chunk start ERROR: ' + (err.message || String(err)));
      return;
    }

    setTimeout(() => {
      try {
        if (chunkRecorder && chunkRecorder.state === 'recording') chunkRecorder.stop();
      } catch (err) {}
    }, ms);
  }

  function blobToBase64(blob) {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onloadend = () => {
        const dataUrl = String(reader.result || '');
        resolve(dataUrl.split(',')[1] || '');
      };
      reader.onerror = reject;
      reader.readAsDataURL(blob);
    });
  }

  function fileToBase64(file) {
    return new Promise((resolve, reject) => {
      const reader = new FileReader();
      reader.onloadend = () => {
        const dataUrl = String(reader.result || '');
        resolve(dataUrl.split(',')[1] || '');
      };
      reader.onerror = reject;
      reader.readAsDataURL(file);
    });
  }


  function appendTranscript(text) {
    const clean = String(text || '').trim();
    if (!clean) return;
    if (transcriptBox.value && !transcriptBox.value.endsWith('\n')) transcriptBox.value += '\n';
    transcriptBox.value += clean + '\n';
    transcriptBox.scrollTop = transcriptBox.scrollHeight;
    updateCounters();
  }

  function appendTranslation(text) {
    const clean = String(text || '').trim();
    if (!clean) return;
    if (translationBox.value && !translationBox.value.endsWith('\n')) translationBox.value += '\n';
    translationBox.value += clean + '\n';
    translationBox.scrollTop = translationBox.scrollHeight;
  }

  function updateCounters() {
    const text = transcriptBox.value.trim();
    charCounter.innerText = text.length + ' ตัวอักษร';
    wordCounter.innerText = (text ? text.split(/\s+/).length : 0) + ' คำ';
  }

  async function stopCapture() {
    const oldSession = session;
    isRecording = false;

    if (chunkTimer) {
      clearInterval(chunkTimer);
      chunkTimer = null;
    }

    try {
      if (chunkRecorder && chunkRecorder.state === 'recording') chunkRecorder.stop();
    } catch (err) {}

    try {
      if (fullRecorder && fullRecorder.state === 'recording') fullRecorder.stop();
    } catch (err) {}

    if (mediaStream) {
      mediaStream.getTracks().forEach(t => t.stop());
      mediaStream = null;
    }

    updateUIState();

    if (oldSession && oldSession.folderId) {
      try {
        if (shouldSaveMedia()) log('กำลังรวมและอัปโหลดไฟล์เสียงเดียว...');
        else log('ข้ามการอัปโหลดไฟล์เสียงตามโหมดบันทึก');
        await waitForFullRecorderStop_();

        let fullAudioBase64 = '';
        if (shouldSaveMedia() && fullAudioChunks.length) {
          const fullBlob = new Blob(fullAudioChunks, { type: fullAudioMimeType || 'audio/webm' });
          log('ไฟล์เสียงเต็ม: ' + Math.round(fullBlob.size / 1024) + ' KB');
          fullAudioBase64 = await blobToBase64(fullBlob);
        }

        const done = await apiPost({
          action: 'finishSession',
          folderId: oldSession.folderId,
          transcriptFileId: oldSession.transcriptFileId,
          translationFileId: oldSession.translationFileId,
          fullAudioBase64,
          fullAudioMimeType: fullAudioMimeType || 'audio/webm',
          saveMode: saveMode.value
        });

        if (done.error) {
          log('finish ERROR: ' + done.error);
          showToast('Upload Error', done.error, 'error');
          return;
        }

        if (done.folderUrl) {
          openFolderLink.href = done.folderUrl;
          openFolderLink.classList.remove('hidden');
          log('เสร็จแล้ว: ' + done.folderUrl);
        }

        if (done.audioFileUrl) {
          log('ไฟล์เสียงเดียว: ' + done.audioFileUrl);
        }

        // ล้างเสียงออกจาก Browser หลังอัปโหลดสำเร็จ
        fullAudioChunks = [];
        showToast('เสร็จแล้ว', 'ดำเนินการตามโหมด: ' + getSaveModeLabel(), 'success');

      } catch (err) {
        log('finish ERROR: ' + (err.message || String(err)));
      }
    }

    log('หยุดถอดเสียง');
  }

  function waitForFullRecorderStop_() {
    return new Promise(resolve => {
      if (!fullRecorder || fullRecorder.state === 'inactive') {
        setTimeout(resolve, 500);
        return;
      }

      const oldOnStop = fullRecorder.onstop;
      fullRecorder.onstop = function(e) {
        if (typeof oldOnStop === 'function') oldOnStop(e);
        setTimeout(resolve, 500);
      };

      try {
        if (fullRecorder.state === 'recording') fullRecorder.stop();
      } catch (err) {
        resolve();
      }
    });
  }

  function copyText(value, label) {
    if (!value.trim()) return;
    const temp = document.createElement('textarea');
    temp.value = value.trim();
    document.body.appendChild(temp);
    temp.select();
    document.execCommand('copy');
    document.body.removeChild(temp);
    showToast('คัดลอกแล้ว', label, 'success');
  }

  function downloadText(filename, text) {
    if (!text.trim()) return;
    const blob = new Blob([text], { type: 'text/plain;charset=utf-8' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = filename;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
  }
